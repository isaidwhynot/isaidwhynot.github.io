			function checkCompanies(node){
				if(node.value.length < 3){
					similarCompaniesPanel.style.display = 'none';
					return;
				}
				$.ajax({
					type:'get',
					data:{cmd:'get_similar_companies',company_name:node.value},
					url:'../ajax/company.php',
					dataType:'JSON',
					success:function(companies){
								if(companies!='-1'){

									similarCompaniesPanel.style.top = ($(node).offset().top + 33)+'px';
									similarCompaniesPanel.style.left = ( $(node).offset().left + $(node).width() - 370)+'px';
									similarCompaniesPanel.innerHTML = '';
									

									for(i=0;i<companies.length;i++){
										similarCompaniesPanel.innerHTML += '<div><a href="?page=company&content=detail&CompanyID='+companies[i].CompanyID+'">'+companies[i].Name+'</a></div>';

									}
									if(companies.length>0){
										similarCompaniesPanel.style.display = 'block';
									}
									else
										similarCompaniesPanel.style.display = 'none';
								}
							}
				});
			}
